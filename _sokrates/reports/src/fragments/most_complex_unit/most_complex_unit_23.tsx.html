function toAppUser(u: any): AppUser {
  return {
    id: Number(u.id ?? 0),
    username: u.username ?? "",
    email: u.email ?? "",
    isAdmin: Number(u.isAdmin ?? u.admin ?? 0) === 1,
  };
}
